import os
import speech_recognition as sr
import pandas as pd

# Directory path containing emotion-labeled subdirectories
directory = r"C:/Users/Nagham/Downloads/deep learning/project/emotion"

# Initialize recognizer
recognizer = sr.Recognizer()

data = []

# List all directories (emotions) in the main directory
emotions = os.listdir(directory)
print("Emotions found in the directory:", emotions)

# Iterate through each emotion directory
for emotion in emotions:
    emotion_dir = os.path.join(directory, emotion)
    
    # Check if the path is a directory
    if os.path.isdir(emotion_dir):
        print(f"Processing files in '{emotion}' directory...")
        
        # List all files in the emotion directory
        files = os.listdir(emotion_dir)
        
        # Transcribe each audio file in the emotion directory
        for file in files:
            audio_file = os.path.join(emotion_dir, file)
            
            # Ensure the file is an audio file (adjust extensions if needed)
            if file.endswith(".wav"):  # Assuming audio files are in .wav format
                with sr.AudioFile(audio_file) as source:
                    audio_data = recognizer.record(source)  # Record the entire audio file
                    
                try:
                    text = recognizer.recognize_google(audio_data, language="ar-EG")
                    print(f"Transcription for '{file}': {text}")
                    
                    # Append emotion, file name, and transcription to the data list
                    data.append({"Emotion": emotion, "File": file, "Transcription": text})
                    
                except sr.UnknownValueError:
                    print(f"Could not understand '{file}'")
                except sr.RequestError as e:
                    print(f"Error: {e}")
                    
        print(f"Finished processing '{emotion}' directory.")

# Create a DataFrame from the collected data
df = pd.DataFrame(data)

# Save the DataFrame to an Excel file
excel_file_path = "transcriptions.xlsx"
df.to_excel(excel_file_path, index=False)

print(f"Data exported to '{excel_file_path}'")

import os
import librosa
import pandas as pd

# Directory path containing emotion-labeled subdirectories
directory = r"C:/Users/Nagham/Downloads/deep learning/project/emotion"

data = []

# List all directories (emotions) in the main directory
emotions = os.listdir(directory)
print("Emotions found in the directory:", emotions)

# Iterate through each emotion directory
for emotion in emotions:
    emotion_dir = os.path.join(directory, emotion)
    
    # Check if the path is a directory
    if os.path.isdir(emotion_dir):
        print(f"Processing files in '{emotion}' directory...")
        
        # List all files in the emotion directory
        files = os.listdir(emotion_dir)
        
        # Extract features for each audio file in the emotion directory
        for file in files:
            audio_file = os.path.join(emotion_dir, file)
            
            # Ensure the file is an audio file (adjust extensions if needed)
            if file.endswith(".wav"):  # Assuming audio files are in .wav format
                try:
                    # Load audio file using Librosa
                    y, sr = librosa.load(audio_file)
                    
                    # Extract MFCCs
                    mfccs = librosa.feature.mfcc(y=y, sr=sr, n_mfcc=13)
                    
                    # Compute mean of MFCCs
                    mfccs_mean = mfccs.mean(axis=1)
                    
                    # Append emotion, file name, and MFCCs to the data list
                    data.append({"Emotion": emotion, "File": file, "MFCCs": mfccs_mean})
                    
                except Exception as e:
                    print(f"Error processing '{file}': {e}")
                    
        print(f"Finished processing '{emotion}' directory.")

# Create a DataFrame from the collected data
df = pd.DataFrame(data)

# Save the DataFrame to a CSV file
csv_file_path = "audio_features.csv"
df.to_csv(csv_file_path, index=False)

print(f"Data exported to '{csv_file_path}'")
